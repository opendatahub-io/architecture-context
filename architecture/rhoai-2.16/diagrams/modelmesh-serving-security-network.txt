Service Mesh Configuration
═══════════════════════════════════════════════════════════════════════════════════

PeerAuthentication (recommended for production):
  name: modelmesh-serving
  namespace: {user-namespace}
  spec:
    mode: STRICT  # or PERMISSIVE for gradual rollout
    selector:
      matchLabels:
        modelmesh-service: {predictor-name}
  # Enforces mTLS for all traffic to/from ModelMesh serving pods

AuthorizationPolicy:
  name: modelmesh-controller-access
  namespace: {user-namespace}
  spec:
    selector:
      matchLabels:
        modelmesh-service: {predictor-name}
    action: ALLOW
    rules:
    - from:
      - source:
          principals:
          - "cluster.local/ns/{controller-namespace}/sa/modelmesh-controller"
      - source:
          namespaces: ["{user-namespace}"]
      to:
      - operation:
          methods: ["GET", "POST"]
          paths: ["/inference.GRPCInferenceService/*", "/v2/models/*/infer"]

RequestAuthentication (optional - for user auth):
  name: modelmesh-jwt-auth
  namespace: {user-namespace}
  spec:
    selector:
      matchLabels:
        modelmesh-service: {predictor-name}
    jwtRules:
    - issuer: "{your-oauth-provider}"
      jwksUri: "{jwks-endpoint}"
