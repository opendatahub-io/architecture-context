NETWORK POLICIES:
- default-allow-same-namespace:
  - Pod Selector: All pods
  - Ingress From: Same namespace pods
  - Ports: All
  - Purpose: Allow intra-namespace communication

- seaweedfs:
  - Pod Selector: app=seaweedfs
  - Ingress From: Namespaces with app.kubernetes.io/part-of=kubeflow-profile;
                  istio-system namespace
  - Ports: 8333/TCP
  - Purpose: Restrict SeaweedFS access to pipeline user namespaces
